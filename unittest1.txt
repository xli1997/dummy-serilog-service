using SymbolMappingPublisher;
using Xunit;

namespace SymbolMappingPublisher.Tests
{
    public class SymbolMappingTests
    {
        [Fact]
        public void ItemSymbol_GetSet_Success()
        {
            // Arrange
            var symbolMapping = new SymbolMapping();
            var testSymbol = "TestSymbol";

            // Act
            symbolMapping.ItemSymbol = testSymbol;
            var result = symbolMapping.ItemSymbol;

            // Assert
            Assert.Equal(testSymbol, result);
        }
    }
}
